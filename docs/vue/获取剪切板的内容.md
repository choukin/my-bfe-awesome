
```html
    <div class="codeCopy" @paste="handlePaste">
      <input v-for="(item, index) in inputList" :key="index" class="border-input" v-model="codeList['code' + index]" maxlength="1" @keyup="nextFocus($event, index)" />
    </div>
```
```js
    // 读取剪切板的内容
    handlePaste: function(event) {
      if (event.clipboardData || event.originalEvent) {
        const clipboardData = (event.clipboardData || event.originalEvent.clipboardData).getData('text')
        if (clipboardData) {
          for (let index = 0; index < clipboardData.length; index++) {
            const element = clipboardData[index]
            this.codeList['code' + index] = element
          }
        }
      }
    }
```
